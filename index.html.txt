        } else {
          let cache = JSON.parse(localStorage.getItem("ongc_ac_cache") || "[]");
          let cachedAsset = cache.find(ac => ac.tag === scannedAsset);
          
          if (cachedAsset) {
            populateAssetUI({
              found: true,
              make: cachedAsset.make, capacity: cachedAsset.capacity,
              area: cachedAsset.area, fullLoc: cachedAsset.fullLoc,
              lastPMDate: cachedAsset.lastPMDate, nextPMDate: cachedAsset.nextPMDate,
              hasOpenComplaint: false, lastRepairDate: "Check online"
            });
          } else {
            alert("Asset not found in offline cache. Connect to internet to download full database.");
            location.reload();
          }
        }
      }

      window.addEventListener('online', updateNetworkStatus);
      window.addEventListener('offline', updateNetworkStatus);

      function updateNetworkStatus() {
        const badge = document.getElementById('sync-badge');
        const icon = document.getElementById('sync-icon');
        const text = document.getElementById('sync-text');
        let queue = JSON.parse(localStorage.getItem("ongc_offline_queue") || "[]");

        if (navigator.onLine) {
          if (queue.length > 0) { checkAndSyncOfflineData(); } 
          else { badge.style.display = 'none'; }
        } else {
          badge.style.display = 'flex'; badge.className = 'sync-badge';
          icon.innerText = 'cloud_off'; icon.classList.remove('spin-icon');
          text.innerText = queue.length > 0 ? `Offline (${queue.length} Pending)` : 'Offline Mode';
        }
      }

      async function submitToDatabase() {
        const btn = document.getElementById('submit-btn');
        const name = document.getElementById('input-name').value;
        const desc = document.getElementById('modal-desc').value;

        if (!name || !desc) return alert("Please fill all mandatory fields");

        let extra = ""; let status = (currentAction === 'Breakdown') ? "OPEN" : "CLOSED";

        if (currentAction === 'Breakdown') {
          let cats = []; document.querySelectorAll('input[name="prob-cat"]:checked').forEach(c => cats.push(c.value));
          extra = "Category: " + (cats.join(", ") || "General Issue");
        } else {
          extra = "Job: " + (document.querySelector('input[name="job-type"]:checked')?.value || "Service");
          if (document.querySelector('input[name="job-status"]:checked')?.value.includes("Pending")) status = "PENDING";
        }

        const payload = { tag: scannedAsset, type: currentAction, userName: name, category: extra, description: desc, status: status };
        btn.innerText = "Processing..."; btn.disabled = true;

        if (!navigator.onLine) {
          saveToOfflineQueue(payload, "Saved to device.");
          return;
        }

        try {
          const response = await apiCall("handleInAppSubmission", { payload: payload });
          if (response.success) { showSuccessUI(response.ticket, payload.type); } 
          else { alert("Error: " + response.error); btn.innerText = "SUBMIT"; btn.disabled = false; }
        } catch (error) {
          saveToOfflineQueue(payload, "Connection timed out. Saved locally.");
        }
      }

      function saveToOfflineQueue(data, msg) {
        let queue = JSON.parse(localStorage.getItem("ongc_offline_queue") || "[]");
        data.offlineSavedAt = new Date().toLocaleString();
        queue.push(data);
        localStorage.setItem("ongc_offline_queue", JSON.stringify(queue));
        
        document.getElementById('report-modal').style.display = 'none';
        alert("⚠️ " + msg + "\n\nYour log for " + data.tag + " will sync automatically when you get signal.");
        updateNetworkStatus(); 
        setTimeout(() => { location.reload(); }, 500);
      }

      async function checkAndSyncOfflineData() {
        if (!navigator.onLine) return;

        let queue = JSON.parse(localStorage.getItem("ongc_offline_queue") || "[]");
        if (queue.length === 0) return;

        const badge = document.getElementById('sync-badge');
        badge.style.display = 'flex'; badge.className = 'sync-badge syncing'; 
        document.getElementById('sync-icon').innerText = 'sync';
        document.getElementById('sync-icon').classList.add('spin-icon');
        document.getElementById('sync-text').innerText = `Syncing ${queue.length} logs...`;

        let itemsProcessed = 0; let failedItems = [];

        for (const item of queue) {
          try {
            const res = await apiCall("handleInAppSubmission", { payload: item });
            if (!res.success) failedItems.push(item);
          } catch (e) { failedItems.push(item); }
          itemsProcessed++;
        }
        checkSyncCompletion(itemsProcessed, queue.length, failedItems);
      }

      function checkSyncCompletion(processed, total, failedItems) {
        if (processed === total) {
          if (failedItems.length === 0) {
            localStorage.removeItem("ongc_offline_queue");
            const badge = document.getElementById('sync-badge');
            badge.className = 'sync-badge online-mode'; 
            document.getElementById('sync-icon').classList.remove('spin-icon');
            document.getElementById('sync-icon').innerText = 'check_circle';
            document.getElementById('sync-text').innerText = 'Sync Complete';
            setTimeout(() => { badge.style.display = 'none'; location.reload(); }, 3000);
          } else {
            localStorage.setItem("ongc_offline_queue", JSON.stringify(failedItems));
            updateNetworkStatus();
          }
        }
      }

      async function sendChatMessage() {
        const input = document.getElementById('chat-input'); 
        const userQ = input.value; 
        if(!userQ) return;
        
        const history = document.getElementById('chat-history'); 
        history.innerHTML += `<div class="chat-bubble chat-user">${userQ}</div>`; 
        input.value = ""; 
        history.scrollTop = history.scrollHeight;
        
        const typingId = "msg-" + Date.now();
        history.innerHTML += `<div id="${typingId}" class="chat-bubble chat-ai" style="opacity:0.6;">Thinking...</div>`;
        history.scrollTop = history.scrollHeight;
        
        try {
          const ans = await apiCall("processUserChat", { question: userQ, chatHistory: myChatMemory });
          document.getElementById(typingId).remove();
          history.innerHTML += `<div class="chat-bubble chat-ai">${ans}</div>`; 
          history.scrollTop = history.scrollHeight; 
          myChatMemory += `\nU: ${userQ}\nA: ${ans}`; 
        } catch (error) {
          document.getElementById(typingId).remove();
          history.innerHTML += `<div class="chat-bubble chat-ai" style="color:red;">⚠️ Network/Server Error: Could not reach the AI.</div>`;
          history.scrollTop = history.scrollHeight;
        }
      }
